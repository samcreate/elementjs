1 (function(window) {
  2 	
  3 
  4 	/**
  5 	* //Path class helps draw a path
  6 	* @class Path
  7 	* @extends EventTarget
  8 	* @constructor
  9 	*/
 10 	
 11 	Path = function() {
 12 		
 13 		EventTarget.call(this);
 14 		this._paths = this._paths || [];
 15 		this.extend(this, new Events());
 16 		
 17 	}	
 18 
 19 	var _pt = Path.prototype = new DisplayObject();
 20 	
 21 	_pt.constructor = Path;
 22 	
 23 	_pt.init = function(p_context){
 24 		this._context = p_context;
 25 		this.transform = new Transform(this._context);
 26 
 27 		this.bind('draw',this.draw,this);
 28 
 29 	}
 30 	
 31 	_pt.color = function(p_color){
 32 		
 33 		if(p_color != null){
 34 			this._color = p_color;
 35 			this.dirty(true, "color");
 36 			return this;
 37 		}else{
 38 			return this._color || "#000000";
 39 		}
 40 	
 41 	}
 42 	
 43 	_pt.beginPath = function(p_x,p_y){
 44 		
 45 		this.dirty(true, "beginPath");
 46 		this._p_startX = p_x;
 47 		this._p_startY = p_y;
 48 		
 49 		return this;
 50 	}
 51 	
 52 	_pt.style = function(p_obj){
 53 		
 54 		this.dirty(true, "style");
 55 		if(p_obj.width) this.width(p_obj.width);
 56 		if(p_obj.color) this.color(p_obj.color);
 57 		
 58 		return this;
 59 	}
 60 	
 61 	_pt.to = function(p_x,p_y){
 62 		
 63 		this.dirty(true, "to");
 64 		this._paths.push({x:p_x,y:p_y});
 65 		return this;
 66 	}
 67 	
 68 	_pt.end = function(p_tf){
 69 		
 70 		this.dirty(true, "end");
 71 		if(p_tf != null){
 72 			this._end = p_tf;
 73 			return this;
 74 		}else{
 75 			return this._end || false;
 76 		}
 77 		
 78 	}
 79 	
 80 	_pt.draw = function(p_val){
 81 		
 82 		this.fire("beginDraw");
 83 		this.transform.save();
 84 		this.transform.setMatrix([1, 0, 0, 1, 0, 0]);
 85 		
 86 		
 87 		this.transform.context.beginPath();
 88 		this.transform.context.moveTo(this._p_startX,this._p_startY);
 89 		for (var i=0; i < this._paths.length; i++) {
 90 			if(this._paths[i] != null) this.transform.context.lineTo(this._paths[i].x,this._paths[i].y);
 91 		};
 92 		this.transform.context.strokeStyle   = this.color() || '#000';
 93 		this.transform.context.lineWidth   = this.width() || 1;
 94 
 95 		this.transform.context.globalAlpha = this.alpha();
 96 		this._applyShadow(this.transform.context);
 97 		if(this.end()) this.transform.context.closePath();
 98 		this.transform.context.stroke();
 99 		
100 		this.transform.restore();
101 	
102 		this.dirty(false);
103 		
104 		this.fire("finishDraw");
105 		
106 		
107 		return this;
108 	};
