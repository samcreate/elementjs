 55 		
 56 		var _to_draw, _w = (this.width()/this.orig_width), _h = (this.height()/this.orig_height);
 57 		
 58 		this._transform_reset();
 59 
 60 		this.fire("beginDraw");
 61 
 62 		if(this.trace()){
 63 			_to_draw = this._handle_trace(_w,_h, this._src, "initial");
 64 		}else{
 65 			_to_draw = this._handle_basic(_w,_h);
 66 		}
 67 	
 68 		_to_draw = this._handle_filters(_to_draw);
 69

 71 		
 72 		this._lastDrawState = _to_draw;

 74

 97 	_pt.toDataURL = function(p_area){
 98 
 99 		this._dataurl_canvas = this._dataurl_canvas || Util.createContext("dataurl_"+this.id());
100 		this._dataurl_transform = this._dataurl_transform || new Transform(this._dataurl_canvas.context);
101 		if(!p_area){
102 			this._dataurl_canvas.canvas.width = this.width();
103 			this._dataurl_canvas.canvas.height = this.height();
104 		}
105 		this._dataurl_transform.save();
106 		this._dataurl_transform.setMatrix([1, 0, 0, 1, 0, 0]);
107 		this._dataurl_canvas.context.clearRect(0, 0, this._dataurl_canvas.canvas.width, this._dataurl_canvas.canvas.height );
108 		this._dataurl_transform.translate(this.x(),this.y());
109 		var _w = (this.width()/this.orig_width),
110 		_h = (this.height()/this.orig_height),
111 		_m2 = this._dataurl_transform.getMatrix();
112 		
113 		if(p_area != null){
114 			this._dataurl_canvas.context.drawImage(window.$$_canvas.canvas(),0,0);
115 		}else{
116 			this._dataurl_transform.rotate(this.rotate());
117 			this._dataurl_transform.scale((this.width()/this.orig_width)*this.scale(),(this.height()/this.orig_height)*this.scale());
118 			this._dataurl_transform.context.globalAlpha = this.alpha();
119 			this._applyShadow(this._dataurl_transform.context);
120 			this._dataurl_canvas.context.drawImage(this._lastDrawState,0,0);
121 		}
122 		
123 		return this._dataurl_canvas.canvas.toDataURL();
124 	};

164 		
165 		(function(_this,_path){
166 			var _img = new Image();
167 			_img.src = _path;
168 			
169 			_img.onload = function() {

173 				_this.width( _this.width() ||  _img.width);
174 				_this.height( _this.height() || _img.height );
175 				_this._src = _img;
176 				_this.fire("loaded");

184 	_pt._handle_filters = function(p_disp_data){
185 		var _to_draw;
186 		if(this.filter() && !this.filter_cache){
187 			// if filter is present and there's no cached filter, let's create one.
188 			if(this._filter_canvas) Util.removeElement(this._filter_canvas.canvas.id);
189 			this._filter_canvas = this._filter_canvas || Util.createContext("filter_"+this.id());
190 			this._filter_canvas.canvas.width = this.orig_width;
191 			this._filter_canvas.canvas.height = this.orig_height;
192 			if(this.trace()){
193 				this._filter_canvas.context.drawImage(p_disp_data,-this.x(),-this.y());
194 			}else{
195 				this._filter_canvas.context.drawImage(p_disp_data,0,0);
196 			}
197 			
198 			for(var key in  this.filter()){
199 				
200 				_to_draw = Filter.get(this._filter_canvas,this.filter()[key],this);
201 				this.filter_cache = _to_draw;
202 			}
203 			//
204 		}else if(this.filter_cache){
205 			//if filter cahce is present let's use it
206 			_to_draw = this.filter_cache;
207 			
208 		}else{
209 			//if filter isn't on, let's set the variable to parameter passed in
210 			_to_draw =  p_disp_data;
211 		}
212 		
213 		//if trace is on, let's create a trace canvas for the filter.
214 		if(this.trace() && this.filter()){
215 			_to_draw = this._handle_trace(1,1, _to_draw, "filter");
216 		}
217 		
218 		return _to_draw;
219 	};

221 	_pt._handle_trace = function(p_w, p_h, p_src, p_key){
222 		//traceing the animation
223 		var _trace_canvas = this["trace_"+this.id()+"_"+p_key] || Util.createContext("trace_"+this.id()+"_"+p_key);
224 		this["trace_"+this.id()+"_"+p_key] = _trace_canvas;
225 		var _trace_transform = this["trace_transfrom_"+this.id()+"_"+p_key] || new Transform(_trace_canvas.context);
226 		this["trace_transfrom_"+this.id()+"_"+p_key] = _trace_transform;
227 		_trace_transform.save();
228 		_trace_transform.setMatrix([1, 0, 0, 1, 0, 0]);
229 		_trace_transform.translate(this.x(),this.y());
230 		_trace_transform.rotate(this.rotate());
231 		_trace_transform.scale(p_w*this.scale(),p_h*this.scale());
232 		_trace_transform.context.globalAlpha = this.alpha();
233 		this._applyStroke(_trace_canvas.context,{x:0,y:0,w:this.orig_width,h:this.orig_height});
234 		this._applyShadow(_trace_transform.context);
235 		_trace_canvas.context.drawImage(p_src,0,0);
236 		return _trace_canvas.canvas;
237 	};

244 		this.transform.translate(this.x(),this.y());

247

249 		return this._src;
