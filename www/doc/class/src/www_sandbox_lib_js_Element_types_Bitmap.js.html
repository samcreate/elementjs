 59 	_pt.draw = function(){
 60 		
 61 		var _to_draw, _w = (this.width()/this.orig_width), _h = (this.height()/this.orig_height);
 62 		
 63 		this._transform_reset();
 64 
 65 		this.fire("beginDraw");
 66 
 67 		if(this.trace()){
 68 			_to_draw = this._handle_trace(_w,_h, this._src, "initial");
 69 		}else{
 70 			_to_draw = this._handle_basic(_w,_h);
 71 			
 72 		}
 73 
 74 		
 75 		 _to_draw = this._handle_filters(_to_draw);
 76 
 77 		 if(this.mask()){
 78 			this.transform.restore();
 79 			 _to_draw = this._handle_mask(_to_draw);
 80 		
 81 		}
 82 		this.transform.context.drawImage(_to_draw,0,0);
 83 		this._lastDrawState = _to_draw;
 84 	
 85 		this.transform.restore();
 86 
 87 		this.dirty(false);
 88 		
 89 		this.fire("finishDraw");
 90 
 91 
 92 		
 93 		return this;
 94 	};

 96 	_pt._handle_mask =  function(p_disp_data){
 97 
 98 		this._mask_canvas = this._mask_canvas || Util.createContext("mask_"+this.id());
 99 		this._mask_transform = this._mask_transform || new Transform(this._mask_canvas.context);
100 		this._mask_transform.save();
101 		this._mask_transform.setMatrix([1, 0, 0, 1, 0, 0]);
102 		this._mask_canvas.context.clearRect(0, 0, this._mask_canvas.canvas.width, this._mask_canvas.canvas.height );
103 		this._mask_transform.context.globalAlpha = this.alpha();
104 		this._applyStroke(this._mask_transform.context,{x:this.x(),y:this.y(),w:this.width(),h:this.height()});
105 		this._mask_transform.translate(this.x(),this.y());
106 		this._mask_transform.rotate(this.rotate());
107 		this.transform.rotate(this.rotate());
108 		this._mask_transform.scale((this.width()/this.orig_width)*this.scale(),(this.height()/this.orig_height)*this.scale());		
109 		this.mask()._make(this._mask_canvas.context,this);
110 		this._mask_canvas.context.drawImage(p_disp_data,0,0);
111 
112 		return this._mask_canvas.canvas;
113 	};

119     * @description 
120     // toDataURL pulls the image data in a dataURL format. This function allows for 
121     // one optional string parameter. If you pass in "all", all of the image data
122     // is returned for the entire canvas, if left blank, just that objects data is

124 	* @public

126 	// this example will return dataURL for just that single object
127 	*
128    	*var _img_data = new Element("Bitmap",{
129 	*		src:"path/to/image.jpg"
130 	*}).toDataURL();
131 	*
132 	//this example will return all image data on the "main" canvas element

134    	*var _img_data = new Element("Bitmap",{
135 	*		src:"path/to/image.jpg"
136 	*}).toDataURL("all");

138 	_pt.toDataURL = function(p_area){
139 
140 		this._dataurl_canvas = this._dataurl_canvas || Util.createContext("dataurl_"+this.id());
141 		this._dataurl_transform = this._dataurl_transform || new Transform(this._dataurl_canvas.context);
142 		if(!p_area){
143 			this._dataurl_canvas.canvas.width = this.width();
144 			this._dataurl_canvas.canvas.height = this.height();
145 		}
146 		this._dataurl_transform.save();
147 		this._dataurl_transform.setMatrix([1, 0, 0, 1, 0, 0]);
148 		this._dataurl_canvas.context.clearRect(0, 0, this._dataurl_canvas.canvas.width, this._dataurl_canvas.canvas.height );
149 		this._dataurl_transform.translate(this.x(),this.y());
150 		var _w = (this.width()/this.orig_width),
151 		_h = (this.height()/this.orig_height),
152 		_m2 = this._dataurl_transform.getMatrix();
153 		
154 		if(p_area != null){
155 			this._dataurl_canvas.context.drawImage(window.$$_canvas.canvas(),0,0);
156 		}else{
157 			this._dataurl_transform.rotate(this.rotate());
158 			this._dataurl_transform.scale((this.width()/this.orig_width)*this.scale(),(this.height()/this.orig_height)*this.scale());
159 			this._dataurl_transform.context.globalAlpha = this.alpha();
160 			this._applyShadow(this._dataurl_transform.context);
161 			this._dataurl_canvas.context.drawImage(this._lastDrawState,0,0);
162 		}
163 		
164 		return this._dataurl_canvas.canvas.toDataURL();
165 	};

205 	_pt._loadsrc = function(p_path){
206 		
207 		(function(_this,_path){
208 			var _img = new Image();
209 			_img.src = _path;
210 			
211 			_img.onload = function() {
212 				
213 				_this.orig_width = _img.width;
214 				_this.orig_height = _img.height;
215 				_this.width( _this.width() ||  _img.width);
216 				_this.height( _this.height() || _img.height );
217 				_this._src = _img;
218 
219 				if(_this.visible() == false){
220 					_this.dirty(false);
221 				}else{
222 					_this.dirty(true,"_loadsrc");
223 				}
224 				_this.bind('draw',_this.draw,_this);
225 				_this.ready(true);
226 				_this.fire("loaded");
227 				
228 				
229 			};
230 			
231 		})(this,p_path);

234 	_pt._handle_filters = function(p_disp_data){
235 
236 		var _to_draw;
237 		if(this.filter() && !this.filter_cache){
238 			// if filter is present and there's no cached filter, let's create one.
239 			if(this._filter_canvas) Util.removeElement(this._filter_canvas.canvas.id);
240 			this._filter_canvas = this._filter_canvas || Util.createContext("filter_"+this.id());
241 			this._filter_canvas.canvas.width = this.orig_width;
242 			this._filter_canvas.canvas.height = this.orig_height;
243 			if(this.trace()){
244 				this._filter_canvas.context.drawImage(p_disp_data,-this.x(),-this.y());
245 			}else{
246 				this._filter_canvas.context.drawImage(p_disp_data,0,0);
247 			}
248 			
249 			for(var key in  this.filter()){
250 				
251 				_to_draw = Filter.get(this._filter_canvas,this.filter()[key],this);
252 				this.filter_cache = _to_draw;
253 			}
254 			
255 		}else if(this.filter_cache){
256 			//if filter cahce is present let's use it
257 			_to_draw = this.filter_cache;
258 			
259 		}else{
260 			//if filter isn't on, let's set the variable to parameter passed in
261 			_to_draw =  p_disp_data;
262 		}
263 		
264 		//if trace is on, let's create a trace canvas for the filter.
265 		if(this.trace() && this.filter()){
266 			_to_draw = this._handle_trace(1,1, _to_draw, "filter");
267 		}
268 		
269 		return _to_draw;
270 	};

272 	_pt._handle_trace = function(p_w, p_h, p_src, p_key){
273 		//traceing the animation
274 		var _trace_canvas = this["trace_"+this.id()+"_"+p_key] || Util.createContext("trace_"+this.id()+"_"+p_key);
275 		this["trace_"+this.id()+"_"+p_key] = _trace_canvas;
276 		var _trace_transform = this["trace_transfrom_"+this.id()+"_"+p_key] || new Transform(_trace_canvas.context);
277 		this["trace_transfrom_"+this.id()+"_"+p_key] = _trace_transform;
278 		_trace_transform.save();
279 		_trace_transform.setMatrix([1, 0, 0, 1, 0, 0]);
280 		_trace_transform.translate(this.x(),this.y());
281 		_trace_transform.rotate(this.rotate());
282 		_trace_transform.scale(p_w*this.scale(),p_h*this.scale());
283 		_trace_transform.context.globalAlpha = this.alpha();
284 		this._applyStroke(_trace_canvas.context,{x:0,y:0,w:this.orig_width,h:this.orig_height});
285 		this._applyShadow(_trace_transform.context);
286 		_trace_canvas.context.drawImage(p_src,0,0);
287 		return _trace_canvas.canvas;
288 	};

295 		this.transform.translate(this.x(),this.y());
296 		this.transform.rotate(this.rotate());
297 		this.transform.scale(p_w*this.scale(),p_h*this.scale());
298 		
299 		this._applyShadow(this.transform.context);
300 		return this._src;
301 	};
