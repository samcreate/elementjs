103     // toDataURL pulls the image data in a dataURL format. This function allows for

121 	_pt.toDataURL = function(p_area){
122 
123 		this._dataurl_canvas = this._dataurl_canvas || Util.createContext("dataurl_"+this.id(),this.scene().canvas());
124 		this._dataurl_transform = this._dataurl_transform || new Transform(this._dataurl_canvas.context);
125 		if(!p_area){
126 			this._dataurl_canvas.canvas.width = this.width();
127 			this._dataurl_canvas.canvas.height = this.height();
128 		}
129 		this._dataurl_transform.save();
130 		this._dataurl_transform.setMatrix([1, 0, 0, 1, 0, 0]);
131 		this._dataurl_canvas.context.clearRect(0, 0, this._dataurl_canvas.canvas.width, this._dataurl_canvas.canvas.height );
132 		this._dataurl_transform.translate(this.x(),this.y());
133 		var _w = (this.width()/this.orig_width),
134 		_h = (this.height()/this.orig_height),
135 		_m2 = this._dataurl_transform.getMatrix();
136 		
137 		if(p_area != null){
138 			this._dataurl_canvas.context.drawImage(window.$$_canvas.canvas(),0,0);
139 		}else{
140 			this._dataurl_transform.rotate(this.rotate());
141 			this._dataurl_transform.scale((this.width()/this.orig_width)*this.scale(),(this.height()/this.orig_height)*this.scale());
142 			this._dataurl_transform.context.globalAlpha = this.alpha();
143 			this._applyShadow(this._dataurl_transform.context);
144 			this._dataurl_canvas.context.drawImage(this._lastDrawState,0,0);
145 		}
146 		
147 		return this._dataurl_canvas.canvas.toDataURL();
148 	};
