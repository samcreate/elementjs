  6 	* //Sprite class helps draw a Sprite
  7 	* @class Sprite

 12 	Sprite = function(p_context) {
 13 
 14 		EventTarget.call(this);
 15 		this._context = p_context;
 16 		this.extend(this, new Events());
 17 		this.extend(this, new Animate());
 18 		this.transform = new Transform(this._context);
 19 		this.iterate = 0;
 20 		this._sequences = {};
 21 		this._curSequence = null;
 22 		this._stopindex = null;
 23 		this._lastDrawState = "";
 24 	};

 26 	var _pt = Sprite.prototype = new Bitmap();
 27 
 28 	_pt.constructor = Sprite;

 30 	_pt.src = function(p_sprite, p_frames) {
 31 		if (p_sprite && p_frames) {
 32 			Util.getJSON(p_frames, function(p_response) {
 33 				this._src = p_sprite;
 34 
 35 				this.frames = p_response.frames;
 36 				this._curSequence = {
 37 					start: 0,
 38 					end: this.frames.length
 39 				};
 40 				this._loadsrc(this._src);
 41 			}, this);
 42 			return this;
 43 		} else {
 44 			return this._src;
 45 		}
 46 	};

 49 	_pt.draw = function(p_scope) {
 50 
 51 		if(!this.ready()) return;
 52 		
 53 		var _to_draw, _w = (this.width() / this.orig_width), _h = (this.height() / this.orig_height);
 54 
 55 		this._transform_reset();
 56 		
 57 		this.fire("beginDraw");
 58 
 59 		_to_draw = this._handle_sprite(this._src);
 60 
 61 
 62 		if (this.trace()) {
 63 			_to_draw = this._handle_trace(_w,_h,  _to_draw, "initial");
 64 		} else {
 65 			this._handle_basic(_w, _h);
 66 		}
 67 		this.resetFilter();
 68 		_to_draw = this._handle_filters(_to_draw);
 69 
 70 		this.transform.context.drawImage(_to_draw, 0, 0);
 71 	
 72 		this._lastDrawState = _to_draw;
 73 
 74 		this.transform.restore();
 75 		this.fire("finishDraw");
 76 
 77 		return this;
 78 	};

 83 	_pt._handle_sprite = function(p_src) {
 84 		var _to_draw, _w, _h, _x, _y, _ctx, _cvs;
 85 		this._sprite_canvas = this._sprite_canvas || Util.createContext("sprite_" + this.id());
 86 		if (this.iterate < this._curSequence.end - 1) {
 87 			if (this.gotoAndStop() !== null) {
 88 				this.iterate = this.gotoAndStop();
 89 			} else {
 90 				this.iterate++;
 91 			}
 92 
 93 		}
 94 
 95 		_w = this.frames[this.iterate].frame.w + this.padding();
 96 		_h = this.frames[this.iterate].frame.h + this.padding();
 97 		_x = this.frames[this.iterate].frame.x;
 98 		_y = this.frames[this.iterate].frame.y;
 99 
100 		_ctx = this._sprite_canvas.context;
101 		_cvs = this._sprite_canvas.canvas;
102 		_cvs.width = _w;
103 		_cvs.height = _h;
104 
105 		if (!this._src_sprite) {
106 			this._src_sprite = this._src;
107 			this._src = _cvs;
108 		}
109 
110 		_ctx.clearRect(0, 0, _cvs.width, _cvs.height);
111 
112 		_to_draw = _ctx.drawImage(this._src_sprite, -_x, -_y);
113 
114 		if (this.iterate === this._curSequence.end - 1 && this.loop()) {
115 			this.iterate = this._curSequence.start;
116 
117 		}
118 
119 		return _cvs;
120 	};

122 	_pt.loop = function(p_val) {
123 		if (p_val != null) {
124 			this._loop = p_val;
125 			return this;
126 		} else {
127 			return this._loop || false;
128 		}
129 	};

131 	_pt.padding = function(p_val) {
132 		if (p_val != null) {
133 			this._padding = p_val;
134 			return this;
135 		} else {
136 			return this._padding || 0;
137 		}
138 	};
139 
140 	_pt.gotoAndStop = function(p_val) {
141 
142 		if (p_val != null) {
143 			if (typeof p_val === "string") {
144 				this._stopindex = this._sequences[p_val].end;
145 
146 			} else {
147 				this._curSequenceName = "";
148 				this._stopindex = p_val;
149 			}
150 			this.draw();
151 			return this;
152 		} else {
153 			return this._stopindex;
154 		}
155 
156 	};
157 
158 	_pt.gotoAndPlay = function(p_val) {
159 		if (p_val != null) {
160 			if (typeof p_val === "string") {
161 				this.run(p_val);
162 			} else {
163 				this.iterate = p_val;
164 			}
165 			this._stopindex = null;
166 			this.draw();
167 			return this;
168 		} else {
169 			return this._stopindex;
170 		}
171 		return this;
172 	};
173 
174 	_pt.sequence = function(p_name, p_start, p_end) {
175 		this._sequences[p_name] = {
176 			start: p_start,
177 			end: p_end
178 		};
179 		return this;
180 	};
181 
182 	_pt.run = function(p_name, p_loop) {
183 
184 		if (this._curSequenceName != p_name) {
185 
186 			var _seq = this._sequences[p_name];
187 			
188 			_seq.start = parseFloat(_seq.start);
189 			_seq.end = parseFloat(_seq.end);
190 			this._curSequence = _seq;
191 			this._stopindex = null;
192 			if (this._curSequence.start != this._sequences[p_name].start || this._curSequence.start === 0) {
193 
194 				this.iterate = _seq.start;
195 				this._curSequenceName = p_name;
196 
197 			}
198 
199 			
200 			if (p_loop === true) this.loop(true);
201 		}
202 		this.draw();
203 		return this;
204 	};
