  4 (function(window) {
  5 	
  6 	/**
  7 	* //EventTarget class is responsible for all objects event handeling and firing.
  8 	* @class EventTarget
  9 	* @extends Basic
 10 	* @constructor
 11 	* @example var EventTarget = new EventTarget();
 12 	*/
 13 
 14 
 15 	EventTarget = function() {
 16 		
 17 		this._listeners = {};
 18 	};
 19 
 20 
 21 
 22 	var _pt = EventTarget.prototype = new Basic();
 23 	
 24 	_pt.constructor = EventTarget;
 25 
 26 	// ====================
 27 	// = Getter / Setters =
 28 	// ====================
 29 
 30 	
 31 	// ====================
 32 	// = public functions =
 33 	// ====================
 34 	_pt.bind = function(type, listener, scope){
 35 		if (typeof this._listeners[type] == "undefined"){
 36             this._listeners[type] = [];
 37         }
 38 
 39         this._listeners[type].push({fn:listener,scope:scope});
 40 
 41         return this;
 42 	};
 43 	/** @memberOf EventTarget#
 44     * @function
 45     * @description Removes the event from the event list. 
 46     * @name unbind
 47 	* @public 
 48 	* @param {string} type - unique name given to the custom event to be removed
 49 	* @param {function} listener - the callback function that was originally bound to the event
 50 	* @example _eventEngine.unbind('eventName', _eventHandler );
 51     */
 52 
 53     /**
 54 	* @memberOf EventTarget#
 55 	* @name unbind
 56     * @function
 57 	* @param {string} type - unique name given to the custom event to be removed
 58 	* @param {function} listener - the callback function that was originally bound to the event
 59 	* @param {object} scope - the scope of callback's 'this' property
 60     * @description 
 61     // Removes the event from the event list. 
 62 	* @public 
 63 	* @example 
 64 	// In this example, we remove the event "eventName" binding to our
 65 	// function "_eventHandler".
 66    	*var _my_element.unbind('eventName', _eventHandler );
 67 	*console.log(_my_element.x); //returns 33;
 68     */
 69 	_pt.unbind = function(type, listener){
 70 		if (this._listeners[type] instanceof Array){
 71             var listeners = this._listeners[type];
 72             for (var i=0, len=listeners.length; i < len; i++){
 73                 if (listeners[i].fn === listener){
 74                     listeners.splice(i, 1);
 75                     break;
 76                 }
 77             }
 78         }
 79 	};
 80 	/** @memberOf EventTarget#
 81     * @function
 82     * @description triggers the custom event. 
 83     * @name fire
 84 	* @public 
 85 	* @param {string} type - unique name given to the custom event to be fired
 86 	* @param {Object} [params] - passes parameters to the callback function 
 87 	* @example _eventEngine.fire('eventName','pass me to the callback function');
 88     */
 89 	_pt.fire = function(event,params){
 90 		if (typeof event == "string"){
 91             event = { type: event };
 92         }
 93         if (!event.target){
 94             event.target = this;
 95         }
 96 
 97         if (!event.type){  //falsy
 98             throw new Error("Event object missing 'type' property.");
 99         }
100 		
101 		if(params){
102 			event.params = params;
103 		}
104 		
105         if (this._listeners[event.type] instanceof Array){
106             var listeners = this._listeners[event.type];
107             for (var i=0, len=listeners.length; i < len; i++){
108                 listeners[i].fn.call( listeners[i].scope || this, event);
109             }
110         }
111 	};
112 	
113 
114 	// =====================
115 	// = private functions =
116 	// =====================
117 	
118 
119 	
120 	window.EventTarget = EventTarget;
121 	
122 }(window));
