  4 (function(window) {
  5 	
  6 	/** @memberOf EventTarget#
  7  `   * @function
  8     * @description holds a custom event until the event is fired. 
  9     * @name bind
 10 	* @public 
 11 	* @param {string} type - unique name given to the custom event
 12 	* @param {function} listener - call back function
 13 	* @param {Object} [scope] - if passed it envokes the callback function with the scope of the passed-in param. (scope injection)
 14 	* @example _eventEngine.bind('eventName',function(){console.log('callback!')},window);
 15     */
 16 	EventTarget = function() {
 17 		
 18 		this._listeners = {};
 19 	};
 20 
 21 
 22 
 23 	var _pt = EventTarget.prototype = new Basic();
 24 	
 25 	_pt.constructor = EventTarget;
 26 
 27 	// ====================
 28 	// = Getter / Setters =
 29 	// ====================
 30 
 31 	
 32 	// ====================
 33 	// = public functions =
 34 	// ====================
 35 	_pt.bind = function(type, listener, scope){
 36 		if (typeof this._listeners[type] == "undefined"){
 37             this._listeners[type] = [];
 38         }
 39 
 40         this._listeners[type].push({fn:listener,scope:scope});
 41 
 42         return this;
 43 	};
 44 	/** @memberOf EventTarget#
 45     * @function
 46     * @description Removes the event from the event list. 
 47     * @name unbind
 48 	* @public 
 49 	* @param {string} type - unique name given to the custom event to be removed
 50 	* @param {function} listener - the callback function that was originally bound to the event
 51 	* @example _eventEngine.unbind('eventName', _eventHandler );
 52     */
 53 	_pt.unbind = function(type, listener){
 54 		if (this._listeners[type] instanceof Array){
 55             var listeners = this._listeners[type];
 56             for (var i=0, len=listeners.length; i < len; i++){
 57                 if (listeners[i].fn === listener){
 58                     listeners.splice(i, 1);
 59                     break;
 60                 }
 61             }
 62         }
 63 	};
 64 	/** @memberOf EventTarget#
 65     * @function
 66     * @description triggers the custom event. 
 67     * @name fire
 68 	* @public 
 69 	* @param {string} type - unique name given to the custom event to be fired
 70 	* @param {Object} [params] - passes parameters to the callback function 
 71 	* @example _eventEngine.fire('eventName','pass me to the callback function');
 72     */
 73 	_pt.fire = function(event,params){
 74 		if (typeof event == "string"){
 75             event = { type: event };
 76         }
 77         if (!event.target){
 78             event.target = this;
 79         }
 80 
 81         if (!event.type){  //falsy
 82             throw new Error("Event object missing 'type' property.");
 83         }
 84 		
 85 		if(params){
 86 			event.params = params;
 87 		}
 88 		
 89         if (this._listeners[event.type] instanceof Array){
 90             var listeners = this._listeners[event.type];
 91             for (var i=0, len=listeners.length; i < len; i++){
 92                 listeners[i].fn.call( listeners[i].scope || this, event);
 93             }
 94         }
 95 	};
 96 	
 97 
 98 	// =====================
 99 	// = private functions =
100 	// =====================
101 	
102 
103 	
104 	window.EventTarget = EventTarget;
105 	
106 }(window));
