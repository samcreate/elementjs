187 	* @name trace
188     * @function
189 	* @param {Boolean} [true/false] - true/false
190     * @description 
191     // Trace function tells the object wether or not all previous context drawing 
192     // should be saved, never erasing the objects previous drawn state. If no value
193     // is entered, the function acts like a 'getter' for the object's trace state.
194 	* @public 
195 	* @example 
196 	// this example will turn the trace state to 'on.'

198    	*var _img = new Element("Bitmap",{

223 	/** @memberOf DisplayObject#
224 	* @name collides
225     * @function
226 	* @param {Object} Element - anything that inherits DisplayObject
227 	* @param {function} Callback - the callback function trigger when the objects rectangle bonds overlap.
228     * @description 
229     // Collides is used to find out if two different objects collide. 
230 	* @public 
231 	* @example 
232 	// this example will send out an alert if the two object overlap
233 	*
234    	*_my_element.collides(_other_element,function(){
235 	*		alert("_other_element touches _my_element!");
236 	*});
237     */
238 	_pt.collides = function(p_target,p_callback){
239 		var a = this, b = p_target;
240 		var _collide = a.x() < b.x() + b.width() &&
241 		         a.x() + a.width() > b.x() &&
242 		         a.y() < b.y() + b.height() &&
243 		         a.y() + a.height() > b.y();
244 		if(_collide)p_callback.call(this);
245 	};
