163 	_pt.filter = function(){
164 		
165 		if(arguments != undefined && arguments.length > 0){
166 			this.resetFilter();
167 			this._filters = this._filters || [];
168 			var _temp_filter = {};
169 			_temp_filter.args = [];
170 			_temp_filter.effect = arguments[0];
171 			
172 			for (var i=1; i < arguments.length; i++) {
173 				_temp_filter.args.push(arguments[i]);
174 			}
175 			this._filters[_temp_filter.effect] = _temp_filter;
176 			return this;
177 		}else{
178 			return this._filters;
179 		}
180 	};

182 	_pt.collides = function(p_target,p_callback){
183 		var a = this, b = p_target;
184 		var _collide = a.x() < b.x() + b.width() &&
185 		         a.x() + a.width() > b.x() &&
186 		         a.y() < b.y() + b.height() &&
187 		         a.y() + a.height() > b.y();
188 		if(_collide)p_callback.call(this);
189 	};
