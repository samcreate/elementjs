169 	_pt.filter = function(){
170 		
171 		if(arguments != undefined && arguments.length > 0){
172 			this.resetFilter();
173 			this._filters = this._filters || [];
174 			var _temp_filter = {};
175 			_temp_filter.args = [];
176 			_temp_filter.effect = arguments[0];
177 			
178 			for (var i=1; i < arguments.length; i++) {
179 				_temp_filter.args.push(arguments[i]);
180 			}
181 			this._filters[_temp_filter.effect] = _temp_filter;
182 			this.dirty(true,"filter");
183 			return this;
184 		}else{
185 			return this._filters;
186 		}
187 	};
188 
189 	_pt.filterRemove = function(p_filter){
190 		for(var key in  this.filter()){
191 			if(p_filter === key || p_filter === "all" ){
192 				
193 				function _finishDraw(){
194 					this.resetFilter();
195 					this.unbind("finishDraw",_finishDraw);
196 					this.resetFilter();
197 					delete this.filter()[key];
198 					if( Util.getSize(this.filter()) === 0 ) this._filters = null;
199 					this.dirty(true,"filterRemove");
200 					
201 				}
202 
203 				if(p_filter === "all") {
204 					this._filters = null;
205 					this.resetFilter();
206 					this.dirty(true,"filterRemove");
207 					
208 				}else{
209 					this.bind("finishDraw",_finishDraw);
210 					return this;
211 				}
212 				
213 			}
214 		}
215 
216 		return this;
217 	};

219 	_pt.collides = function(p_target,p_callback){
220 		var a = this, b = p_target;
221 		var _collide = a.x() < b.x() + b.width() &&
222 		         a.x() + a.width() > b.x() &&
223 		         a.y() < b.y() + b.height() &&
224 		         a.y() + a.height() > b.y();
225 		if(_collide)p_callback.call(this);
226 	};
