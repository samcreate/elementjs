102 	_pt.filter = function(){
103 		
104 		if(arguments != undefined && arguments.length > 0){
105 			this.resetFilter();
106 			this._filters = this._filters || [];
107 			var _temp_filter = {};
108 			_temp_filter.args = [];
109 			_temp_filter.effect = arguments[0];
110 			
111 			for (var i=1; i < arguments.length; i++) {
112 				_temp_filter.args.push(arguments[i]);
113 			}
114 			this._filters[_temp_filter.effect] = _temp_filter;
115 			this.dirty(true,"filter");
116 			return this;
117 		}else{
118 			return this._filters;
119 		}
120 	};

122 	/**
123 	* @memberOf Graphics#
124 	* @name filterRemove
125     * @function
126 	* @param {String} value - name of the filter, or 'all' to remove all applied filters.
127     * @description 
128     // removes applied filters by name, or if 'all' is passed in, it removes all 
129     // applied filters from the object.
130     *
131     *
132 	* @public 
133 	* @example 
134 	// In this example, we remove the multiply filter previously applied.
135 	*
136    	*_my_element.filterRemove("multiply"); 
137 	*
138 	//In this example, we remove all the applied filters
139 	*_my_element.filterRemove("all");
140 	
141     */
142 
143 	_pt.filterRemove = function(p_filter){
144 		for(var key in  this.filter()){
145 			if(p_filter === key || p_filter === "all" ){
146 				
147 				function _finishDraw(){
148 					this.resetFilter();
149 					this.unbind("finishDraw",_finishDraw);
150 					this.resetFilter();
151 					delete this.filter()[key];
152 					if( Util.getSize(this.filter()) === 0 ) this._filters = null;
153 					this.dirty(true,"filterRemove");
154 					
155 				}
156 
157 				if(p_filter === "all") {
158 					this._filters = null;
159 					this.resetFilter();
160 					this.dirty(true,"filterRemove");
161 					
162 				}else{
163 					this.bind("finishDraw",_finishDraw);
164 					return this;
165 				}
166 				
167 			}
168 		}
169 
170 		return this;
171 	};
