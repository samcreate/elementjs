102 	_pt.filter = function(){
103 		
104 		if(arguments != undefined && arguments.length > 0){
105 			this.resetFilter();
106 			this._filters = this._filters || [];
107 			var _temp_filter = {};
108 			_temp_filter.args = [];
109 			_temp_filter.effect = arguments[0];
110 			
111 			for (var i=1; i < arguments.length; i++) {
112 				_temp_filter.args.push(arguments[i]);
113 			}
114 			this._filters[_temp_filter.effect] = _temp_filter;
115 			this.dirty(true,"filter");
116 			return this;
117 		}else{
118 			return this._filters;
119 		}
120 	};

126 	* @param {String} value - name of the filter, or 'all' to remove all applied filters.
127     * @description 
128     // removes applied filters by name, or if 'all' is passed in, it removes all 
129     // applied filters from the object.
130     *
131     *
132 	* @public 
133 	* @example 
134 	// In this example, we remove the multiply filter previously applied.
135 	*
136    	*_my_element.filterRemove("multiply"); 
137 	*
138 	//In this example, we remove all the applied filters
139 	*_my_element.filterRemove("all");
140 	
141     */

143 	_pt.filterRemove = function(p_filter){
144 		for(var key in  this.filter()){
145 			if(p_filter === key || p_filter === "all" ){
146 				
147 				function _finishDraw(){
148 					this.resetFilter();
149 					this.unbind("finishDraw",_finishDraw);
150 					this.resetFilter();
151 					delete this.filter()[key];
152 					if( Util.getSize(this.filter()) === 0 ) this._filters = null;
153 					this.dirty(true,"filterRemove");
154 					
155 				}
156 
157 				if(p_filter === "all") {
158 					this._filters = null;
159 					this.resetFilter();
160 					this.dirty(true,"filterRemove");
161 					
162 				}else{
163 					this.bind("finishDraw",_finishDraw);
164 					return this;
165 				}
166 				
167 			}
168 		}
169 
170 		return this;
171 	};

185 	_pt._handle_mask =  function(p_disp_data){
186 
187 		this._mask_canvas = this._mask_canvas || Util.createContext("mask_"+this.id(),this.scene().canvas());
188 		this._mask_transform = this._mask_transform || new Transform(this._mask_canvas.context);
189 		this._mask_transform.save();
190 		this._mask_transform.setMatrix([1, 0, 0, 1, 0, 0]);
191 		this._mask_canvas.context.clearRect(0, 0, this._mask_canvas.canvas.width, this._mask_canvas.canvas.height );
192 		this._mask_transform.context.globalAlpha = this.alpha();
193 		this._applyStroke(this._mask_transform.context,{x:this.x(),y:this.y(),w:this.width(),h:this.height()});
194 		this._mask_transform.translate(this.x(),this.y());
195 		this._mask_transform.rotate(this.rotate());
196 		this.transform.rotate(this.rotate());
197 		this._mask_transform.scale((this.width()/this.orig_width)*this.scale(),(this.height()/this.orig_height)*this.scale());		
198 		this.mask()._make(this._mask_canvas.context,this);
199 		this._mask_canvas.context.drawImage(p_disp_data,0,0);
200 
201 		return this._mask_canvas.canvas;
202 	};

204 	_pt._handle_filters = function(p_disp_data){
205 
206 		var _to_draw;
207 		if(this.filter() && !this.filter_cache){
208 			// if filter is present and there's no cached filter, let's create one.
209 			if(this._filter_canvas) Util.removeElement(this._filter_canvas.canvas.id);
210 			this._filter_canvas = this._filter_canvas || Util.createContext("filter_"+this.id(),this.scene().canvas());
211 			this._filter_canvas.canvas.width = this.orig_width;
212 			this._filter_canvas.canvas.height = this.orig_height;
213 			if(this.trace()){
214 				this._filter_canvas.context.drawImage(p_disp_data,-this.x(),-this.y());
215 			}else{
216 				this._filter_canvas.context.drawImage(p_disp_data,0,0);
217 			}
218 			
219 			for(var key in  this.filter()){
220 				_to_draw = Filter.get(this._filter_canvas,this.filter()[key],this);
221 				this.filter_cache = _to_draw;
222 			}
223 			
224 		}else if(this.filter_cache){
225 			//if filter cahce is present let's use it
226 			_to_draw = this.filter_cache;
227 			
228 		}else{
229 			//if filter isn't on, let's set the variable to parameter passed in
230 			_to_draw =  p_disp_data;
231 		}
232 		
233 		//if trace is on, let's create a trace canvas for the filter.
234 		if(this.trace() && this.filter()){
235 			_to_draw = this._handle_trace(1,1, _to_draw, "filter");
236 		}
237 		
238 		return _to_draw;
239 	};

241 	_pt._handle_trace = function(p_w, p_h, p_src, p_key){
242 		//traceing the animation
243 		var _trace_canvas = this["trace_"+this.id()+"_"+p_key] || Util.createContext(("trace_"+this.id()+"_"+p_key),this.scene().canvas());
244 		this["trace_"+this.id()+"_"+p_key] = _trace_canvas;
245 		var _trace_transform = this["trace_transfrom_"+this.id()+"_"+p_key] || new Transform(_trace_canvas.context);
246 		this["trace_transfrom_"+this.id()+"_"+p_key] = _trace_transform;
247 		_trace_transform.save();
248 		_trace_transform.setMatrix([1, 0, 0, 1, 0, 0]);
249 		_trace_transform.translate(this.x(),this.y());
250 		_trace_transform.rotate(this.rotate());
251 		_trace_transform.scale(p_w*this.scale(),p_h*this.scale());
252 		_trace_transform.context.globalAlpha = this.alpha();
253 		this._applyStroke(_trace_canvas.context,{x:0,y:0,w:this.orig_width,h:this.orig_height});
254 		this._applyShadow(_trace_transform.context);
255 		_trace_canvas.context.drawImage(p_src,0,0);
256 		_trace_transform.restore();
257 		return _trace_canvas.canvas;
258 	};
