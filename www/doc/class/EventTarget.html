EventTarget 

							//EventTarget class is responsible for all objects event handeling and firing.
  • .fire(type, params)
  • .unbind(type, listener, scope)
  • .extend(value, value)

fire( type:string, params:Object:Optional ) 

// triggers the custom event
						
	// In this example, we fire of the event 'eventName' with additonal parameters
	// to be passed to any listeners.
_eventEngine.fire('eventName','pass me to the callback function');

unbind( type:string, listener:function, scope:object ) 

// Removes the event from the event list.
						
	// In this example, we remove the event "eventName" binding to our
	// function "_eventHandler".
var _my_element.unbind('eventName', _eventHandler );

extend( value:Object, value:Object )

// extend allows you to take all the properties for one object and
    // apply them to another.
	// In this example, we inherit the x value of obj2.
var _obj2 = {x:33};
var _my_element.extend(_my_element,_obj2);
console.log(_my_element.x); //returns 33;
