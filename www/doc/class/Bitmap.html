Bitmap 

							//Bitmap class loads jpg/png/gif as a Bitmap Object on the canvas
  • .src(path)
  • .toDataURL(area)

src(path) {String} path

// src allows for one string parameter that points to
// the path of an image returns a Bitmap object.

						
							
								
var _img1 = new Element("Bitmap",{
	src:"path/to/image.jpg"
});

toDataURL(area) {String} area

// toDataURL pulls the image data in a dataURL format. This function allows for one optional string parameter. // If you pass in "all", all of the image data is returned for the entire canvas, if left blank, just that objects // data is returned.

						
							
								
var _img_data = new Element("Bitmap",{
	src:"path/to/image.jpg"
}).toDataURL();
